**directions.json**

```
{
  "title": "Apple Pie",
  "directions": [
    "Roll the pie crust",
    "Make the filling",
    "Bake"
  ]
}
```

**ingredients.json**

```
{
  "title": "Apple Pie",
  "directions": [
    "Roll the pie crust",
    "Make the filling",
    "Bake"
  ],
  "ingredients": [
     {
       "amount": {
         "quantity": 2,
           "unit": null
         },
        "name": "pie crusts"
    }, {
       "amount": {
         "quantity": 1,
           "unit": "tbsp"
        },
        "name": "cinnamon"
    }
  ]
}
```
